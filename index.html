// ── GET: Todas as ações (evita CORS) ── function doGet(e) { const ss = SpreadsheetApp.openById(SHEET_ID); const action = (e.parameter.action || 'sync'); // ── SYNC — retornar todos os dados ── if (action === 'sync') { const data = { crm: getSheetData(ss, 'CRM', 'A2:AR1000'), financeiro: getSheetData(ss, 'FINANCEIRO', 'A6:P500'), clientes: getSheetData(ss, 'CLIENTES', 'A5:K200'), projetos: getSheetData(ss, 'PROJETOS', 'A4:P300'), despesas: getSheetData(ss, 'DESPESAS', 'A4:L300'), marketing: getSheetData(ss, 'MARKETING', 'A4:N200'), contatos: getSheetData(ss, 'CONTATOS', 'A2:R2000'), }; return jsonResponse(data); } // ── GET — leitura de range específico ── if (action === 'get') { const sheet = e.parameter.sheet; const range = e.parameter.range; if (!sheet || !range) return jsonResponse({ error: 'Parâmetros sheet e range obrigatórios' }); const data = getSheetData(ss, sheet, range); return jsonResponse(data); } // ── APPEND — adicionar linhas novas ── if (action === 'append') { const sheetName = e.parameter.sheet; const sheet = ss.getSheetByName(sheetName); if (!sheet) { // Criar aba automaticamente se não existir if (sheetName === 'CONTATOS') { return jsonResponse(_criarAbaContatos(ss, e)); } return jsonResponse({ error: 'Aba não encontrada: ' + sheetName }); } const values = JSON.parse(e.parameter.values || '[]'); if (values && values.length) { sheet.getRange(sheet.getLastRow() + 1, 1, values.length, values[0].length).setValues(values); } return jsonResponse({ ok: true, rows: values.length }); } // ── PUT — atualizar range específico ── if (action === 'put') { const sheetName = e.parameter.sheet; var sheet = ss.getSheetByName(sheetName); if (!sheet) { // Criar aba CONTATOS automaticamente se não existir if (sheetName === 'CONTATOS') { sheet = _criarAbaContatosSheet(ss); } else { return jsonResponse({ error: 'Aba não encontrada: ' + sheetName }); } } const range = e.parameter.range; const values = JSON.parse(e.parameter.values || '[]'); if (values && values.length) { // Para a aba CONTATOS: limpar dados antigos antes de reescrever if (sheetName === 'CONTATOS') { const lastRow = sheet.getLastRow(); if (lastRow > 1) { sheet.getRange(2, 1, lastRow - 1, 18).clearContent(); } } sheet.getRange(range).offset(0, 0, values.length, values[0].length).setValues(values); } return jsonResponse({ ok: true }); } // ── BATCH UPDATE — múltiplas atualizações ── if (action === 'batchUpdate') { const updates = JSON.parse(e.parameter.updates || '[]'); updates.forEach(function(u) { var sheet = ss.getSheetByName(u.sheet); if (sheet && u.values && u.values.length) { sheet.getRange(u.range).offset(0, 0, u.values.length, u.values[0].length).setValues(u.values); } }); return jsonResponse({ ok: true, count: updates.length }); } // ── CALENDAR LIST ── if (action === 'calendarList') { try { var calId = e.parameter.calendarId || 'primary'; var events = Calendar.Events.list(calId, { timeMin: e.parameter.timeMin, timeMax: e.parameter.timeMax, maxResults: parseInt(e.parameter.maxResults) || 50, singleEvents: true, orderBy: 'startTime' }); return jsonResponse(events.items || []); } catch(err) { return jsonResponse({ error: 'Erro Calendar: ' + err.message }); } } // ── CALENDAR CREATE ── if (action === 'calendarCreate') { try { var calId2 = e.parameter.calendarId || 'primary'; var event = Calendar.Events.insert({ summary: e.parameter.summary, description: e.parameter.description || '', start: JSON.parse(e.parameter.start || '{}'), end: JSON.parse(e.parameter.end || '{}') }, calId2); return jsonResponse({ ok: true, id: event.id }); } catch(err) { return jsonResponse({ error: 'Erro ao criar evento: ' + err.message }); } } // ── SETUP — criar aba CONTATOS se não existir ── if (action === 'setupContatos') { var existing = ss.getSheetByName('CONTATOS'); if (existing) return jsonResponse({ ok: true, msg: 'Aba CONTATOS já existe' }); _criarAbaContatosSheet(ss); return jsonResponse({ ok: true, msg: 'Aba CONTATOS criada com sucesso' }); } return jsonResponse({ error: 'Ação desconhecida: ' + action }); } // ── POST: Mantido para compatibilidade ── function doPost(e) { const ss = SpreadsheetApp.openById(SHEET_ID); const body = JSON.parse(e.postData.contents); const action = body.action || ''; if (action === 'append') { var sheet = ss.getSheetByName(body.sheet); if (!sheet) { if (body.sheet === 'CONTATOS') sheet = _criarAbaContatosSheet(ss); else return jsonResponse({ error: 'Aba não encontrada: ' + body.sheet }); } const values = body.values; if (values && values.length) { sheet.getRange(sheet.getLastRow() + 1, 1, values.length, values[0].length).setValues(values); } return jsonResponse({ ok: true, rows: values.length }); } if (action === 'put') { var sheet2 = ss.getSheetByName(body.sheet); if (!sheet2) { if (body.sheet === 'CONTATOS') sheet2 = _criarAbaContatosSheet(ss); else return jsonResponse({ error: 'Aba não encontrada: ' + body.sheet }); } const range = body.range; const values = body.values; if (values && values.length) { if (body.sheet === 'CONTATOS') { var lastRow = sheet2.getLastRow(); if (lastRow > 1) sheet2.getRange(2, 1, lastRow - 1, 18).clearContent(); } sheet2.getRange(range).offset(0, 0, values.length, values[0].length).setValues(values); } return jsonResponse({ ok: true }); } if (action === 'batchUpdate') { const updates = body.updates; updates.forEach(function(u) { var sheet = ss.getSheetByName(u.sheet); if (sheet && u.values && u.values.length) { sheet.getRange(u.range).offset(0, 0, u.values.length, u.values[0].length).setValues(u.values); } }); return jsonResponse({ ok: true, count: updates.length }); } if (action === 'calendarList') { try { var calId = body.calendarId || 'primary'; var events = Calendar.Events.list(calId, { timeMin: body.timeMin, timeMax: body.timeMax, maxResults: body.maxResults || 50, singleEvents: true, orderBy: 'startTime' }); return jsonResponse(events.items || []); } catch(err) { return jsonResponse({ error: 'Erro Calendar: ' + err.message }); } } if (action === 'calendarCreate') { try { var calId2 = body.calendarId || 'primary'; var event = Calendar.Events.insert({ summary: body.summary, description: body.description || '', start: body.start, end: body.end }, calId2); return jsonResponse({ ok: true, id: event.id }); } catch(err) { return jsonResponse({ error: 'Erro ao criar evento: ' + err.message }); } } return jsonResponse({ error: 'Ação desconhecida: ' + action }); } // ══════════════════════════════════════════════════════════ // HELPERS // ══════════════════════════════════════════════════════════ function getSheetData(ss, sheetName, range) { var sheet = ss.getSheetByName(sheetName); if (!sheet) return []; try { var data = sheet.getRange(range).getValues(); return data.filter(function(row) { return row.some(function(cell) { return cell !== '' && cell !== null; }); }); } catch(e) { return []; } } function jsonResponse(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } // ── Criar aba CONTATOS com cabeçalho ── function _criarAbaContatosSheet(ss) { var sheet = ss.insertSheet('CONTATOS'); var header = [ 'tipo', // A — proposta | cadastro | externo 'subtipo', // B — projetistas | representantes | outros | setor 'ref_id', // C — ID da proposta (para tipo=proposta) 'nome', // D 'cargo', // E 'area', // F — Geral, Comercial, Financeiro, etc. 'telefone', // G 'email', // H 'doc', // I — CPF/CNPJ (para externos) 'uf', // J 'cidade', // K 'atividade', // L 'esp', // M — especialidade 'valor', // N — valor/taxa (representantes) 'cat', // O — categoria 'custo', // P 'obs', // Q 'data_criacao' // R ]; sheet.getRange(1, 1, 1, header.length).setValues([header]); // Formatar cabeçalho var headerRange = sheet.getRange(1, 1, 1, header.length); headerRange.setFontWeight('bold'); headerRange.setBackground('#1a1a2e'); headerRange.setFontColor('#e0c97f'); sheet.setFrozenRows(1); // Largura das colunas principais sheet.setColumnWidth(1, 90); // tipo sheet.setColumnWidth(2, 120); // subtipo sheet.setColumnWidth(3, 70); // ref_id sheet.setColumnWidth(4, 200); // nome sheet.setColumnWidth(7, 140); // telefone sheet.setColumnWidth(8, 200); // email return sheet; } // Auxiliar para criação via append function _criarAbaContatos(ss, e) { var sheet = _criarAbaContatosSheet(ss); var values = JSON.parse(e.parameter.values || '[]'); if (values && values.length) { sheet.getRange(2, 1, values.length, values[0].length).setValues(values); } return { ok: true, rows: values.length, msg: 'Aba CONTATOS criada e dados inseridos' }; } // ══════════════════════════════════════════════════════════ // SETUP — Executar manualmente uma vez para criar a aba // Menu: Executar → setupContatos // ══════════════════════════════════════════════════════════ function setupContatos() { var ss = SpreadsheetApp.openById(SHEET_ID); var existing = ss.getSheetByName('CONTATOS'); if (existing) { SpreadsheetApp.getUi().alert('A aba CONTATOS já existe!'); return; } _criarAbaContatosSheet(ss); SpreadsheetApp.getUi().alert('✅ Aba CONTATOS criada com sucesso!'); }
